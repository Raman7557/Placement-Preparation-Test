import java.util.Scanner;

class Main {
    public static void main(String[] args) {
      System.out.println("Enter The No.");
      Scanner sc=new Scanner(System.in);
       int a=sc.nextInt();
       int a1;
     a1=a;
     
       
       int temp=0; int rev=0;
       while(a !=0)
       {
           temp=a%10;
          rev=rev*10+temp;
          a=a/10;
       }
      if(rev==a1)
        System.out.println("Entered no is palindrome "+a1);
        else
        System.out.println("It is Not a Palindrome no.");
      
    }
}

import java.util.Scanner;

class Main {
    public static void main(String[] args) {
      System.out.println("Enter The no.");
      Scanner sc=new Scanner(System.in);
       int n=sc.nextInt();
       int a = 0, b = 1;
       

        if (n >= 1) 
            System.out.print(a + " ");
        if (n >= 2)
            System.out.print(b + " ");

        for (int i = 3; i <= n; i++) {
            int c = a + b;
            System.out.print(c + " ");
            a = b;
            b = c;
        }
      
     
      
    }
}
